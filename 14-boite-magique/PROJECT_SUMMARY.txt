📦 PROJECT OVERVIEW
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

A cross-platform C++ tool built with CMake for collecting and archiving 
workshop source files and documents. Works seamlessly on Linux and Windows.

🎯 CORE FEATURES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✓ Recursive directory scanning
✓ File filtering by extensions
✓ Archive creation with structure preservation
✓ Statistics by file type
✓ Cross-platform (Linux/Windows/macOS)
✓ No external dependencies
✓ CLI interface
✓ Automatic exclusions (.git, build)

💡 USAGE EXAMPLES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

# Show statistics
magic-box --stats

# Archive C++ project
magic-box -r ~/project -o ~/backup -e .cpp,.h,.hpp

# Collect documentation
magic-box -r ~/workshop -o ~/docs -e .md,.txt

# Multi-language backup
magic-box -e .cpp,.py,.js,.md -o ./archive
